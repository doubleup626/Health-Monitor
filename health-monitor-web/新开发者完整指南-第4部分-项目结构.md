#### 2. app-service/data-collector.js（⭐核心：数据采集）

**功能**: 在手表后台运行，每5分钟自动采集一次传感器数据

**关键代码结构**:

```javascript
import { BaseSideService } from '@zos/zml/base-side'
import { Time } from '@zos/sensor'

// 采集间隔（分钟）
const COLLECT_INTERVAL_MINUTES = 5

AppService({
  onInit() {
    console.log('App Service onInit')
  },

  onRun() {
    // 创建定时器传感器
    this.timeSensor = new Time()
    
    // 每分钟检查一次
    this.timeSensor.onPerMinute(() => {
      const now = new Date()
      const minute = now.getMinutes()
      
      // 如果当前分钟是5的倍数，执行采集
      if (minute % COLLECT_INTERVAL_MINUTES === 0) {
        this.collectAndSend()
      }
    })
    
    // 立即执行一次采集
    this.collectAndSend()
  },

  collectAndSend() {
    // 采集所有传感器数据
    const healthData = this.collectAllData()
    
    // 通过BLE发送到手机端
    this.request({
      method: 'data.send',
      params: {
        dataList: [healthData]
      }
    })
  },

  collectAllData() {
    // 读取心率、血氧、压力、步数等传感器
    // 返回完整的健康数据对象
    return {
      timestamp: Date.now(),
      deviceId: hmSetting.getDeviceInfo().deviceId,
      heartRate: { ... },
      bloodOxygen: { ... },
      stress: { ... },
      // ...
    }
  },

  onDestroy() {
    // 停止定时器
    this.timeSensor && this.timeSensor.destroy()
  }
})
```

**核心逻辑**:
1. `onRun()`: 启动定时器，每分钟检查一次
2. 每5分钟（分钟数是5的倍数）触发采集
3. `collectAllData()`: 读取所有传感器
4. `request()`: 通过BLE发送到手机端

#### 3. app-side/index.js（⭐核心：数据转发）

**功能**: 在手机端运行，接收手表数据并转发到服务器

**关键代码结构**:

```javascript
import { BaseSideService } from '@zos/zml/base-side'

// 服务器地址 ⚠️ 新电脑必须修改这里
const SERVER_URL = "http://192.168.31.206:3000/api/health-data"

AppSideService({
  onInit() {
    console.log('Side Service onInit')
  },

  onRun() {
    console.log('Side Service onRun')
  },

  // 接收手表端的请求
  onRequest(req, res) {
    console.log('收到手表数据:', req.body)
    
    const params = req.body.params
    
    // 构建请求数据
    const requestData = {
      type: 'batch',
      count: params.dataList.length,
      worker_id: 'W001',  // 可选：工人ID
      data: params.dataList
    }
    
    // 发送到服务器
    this.request({
      method: 'POST',
      url: SERVER_URL,
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(requestData)
    })
    .then(response => {
      console.log('✅ 数据推送成功')
      res({ result: 'success' })
    })
    .catch(error => {
      console.error('❌ 数据推送失败:', error)
      res({ result: 'failed', error: error.message })
    })
  },

  onDestroy() {
    console.log('Side Service onDestroy')
  }
})
```

**核心逻辑**:
1. `onRequest()`: 接收来自手表的BLE请求
2. 构建符合服务器API的数据格式
3. 通过HTTP POST发送到服务器
4. 返回结果给手表端

```javascript
function checkSafety(data) {
  const alerts = []
  let alertLevel = 0  // 0=安全, 1=正常, 2=警告, 3=危险
  
  // 1. 心率检测
  if (data.heartRate.last > THRESHOLDS.heartRate.danger) {
    alerts.push({
      type: 'heart_rate_danger',
      level: 3,
      message: '⚠️ 心率极高：145 bpm（危险！）'
    })
    alertLevel = 3
  }
  
  // 2. 血氧检测
  // 3. 压力检测
  // 4. 深度检测
  // 5. 体温检测
  
  return {
    isSafe: alertLevel === 0,
    alertLevel,
    alertMessage: alerts.map(a => a.message).join('; '),
    alerts
  }
}
```

```javascript
async function performAnalysis(workerData, realtimeData, historyData, safetyCheck) {
  // 1. 构建完整提示词
  const prompt = buildPrompt(workerData, realtimeData, historyData, safetyCheck.alerts)

    },
    body: JSON.stringify({

      messages: [

        { role: 'user', content: prompt }
      ],
      temperature: 0.3,
      max_tokens: 2000
    })
  })
  
  // 3. 解析返回的JSON
  const result = await response.json()
  const analysis = JSON.parse(result.choices[0].message.content)
  
  // 4. 返回分析结果
  return {
    risk_level: analysis.risk_level,
    risk_analysis: analysis.risk_analysis,
    health_status: analysis.health_status,
    safety_advice: analysis.safety_advice,
```

**核心代码**:

```vue
<script setup>
import { ref, onMounted, onUnmounted } from 'vue'

const workers = ref([])
const deviceStats = ref({ online: 0, offline: 0, total: 0 })
let ws = null

// 加载数据
const loadData = async () => {
  const response = await fetch('/api/realtime/all')
  workers.value = await response.json()
}

// WebSocket连接
const connectWebSocket = () => {
  ws = new WebSocket('ws://localhost:3000')
  
  ws.onmessage = (event) => {
    const message = JSON.parse(event.data)
    if (message.type === 'realtime_update') {
      // 更新工人数据
      updateWorkerData(message.worker_id, message.data)
    }
  }
}

// 获取卡片颜色
const getWorkerCardClass = (worker) => {
  if (worker.alert_level === 3) return 'worker-card-danger'
  if (worker.alert_level === 2) return 'worker-card-warning'
  return 'worker-card-safe'
}

onMounted(() => {
  loadData()
  connectWebSocket()
  setInterval(loadData, 5000)  // 每5秒刷新
})

onUnmounted(() => {
  ws && ws.close()
})
</script>

<template>
  <div class="dashboard">
    <!-- 统计卡片 -->
    <div class="stats-grid">
      <el-card>在线: {{ deviceStats.online }}</el-card>
      <el-card>离线: {{ deviceStats.offline }}</el-card>
      <el-card>总数: {{ deviceStats.total }}</el-card>
    </div>
    
    <!-- 工人卡片 -->
    <div class="workers-grid">
      <div v-for="worker in workers" :key="worker.worker_id"
           :class="['worker-card', getWorkerCardClass(worker)]">
        <h3>{{ worker.worker_name }}</h3>
        <div>❤️ {{ worker.heart_rate_last }} bpm</div>
        <div>🫁 {{ worker.blood_oxygen_value }}%</div>
        <div>😰 {{ worker.stress_value }}</div>
        <div>📍 {{ worker.altitude }} m</div>
        <el-button @click="viewDetails(worker.worker_id)">详情</el-button>
        <el-button @click="openAIAnalysis(worker.worker_id)">AI分析</el-button>
      </div>
    </div>
  </div>
</template>
```
